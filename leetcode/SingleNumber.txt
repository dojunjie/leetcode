Given a non-empty array of integers, every element appears twice except for one. Find that single one.
Example 1:

Input: [2,2,1]
Output: 1


class Solution:
    def singleNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
		#用列表会导致时间超时     ？？？
        retList = []
        for i in nums:
            if i in retList:
                retList.remove(i)
            else:
                retList.append(i)
                
        return retList[0]
		
		
		hs = set()
        for n in nums:
            if n in hs:
                hs.remove(n)
            else:
                hs.add(n)
        return hs.pop()
